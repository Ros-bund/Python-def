# объявление функции
def is_prime(num):
    if num < 2:
        return False
    
    # Проверяем делители от 2 до квадратного корня из числа
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False # Найден делитель, число составное
            
    return True # Делителей нет, число простое

# считываем данные
n = int(input())

# вызываем функцию
print(is_prime(n))
